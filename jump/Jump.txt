Jump
	功能 -- 跳转
		startActivity
			隐示跳转
			显示跳转
		startActivityForResult
		sendBroadcastReceiver
		startService

方式: 
	类似 Retrfit2, 注解+动态代理
	
扩展:
	拦截器 仿照OKHttp框架
		JumpInterceptor 跳转拦截
		ParseInterceptor 解析接口参数时拦截 (方法注解和参数注解)
	全局配置: 配置全局拦截器 
	
	解析器:  仿照 Gson 框架 解析参数时拦截
		ParserAdapter <TYPE>{
			int INTENT = 0b01;
			int MEMORY_INTENT = 0b10;
			ParserAdapter(){}
			ParserAdapter(int type){}
			Object putToIntent(TYPE obj){} 
			TYPE parseFromIntent(Object params){}
		}
	
	interface Jump{
		void jumpLogin();//立即跳转
		Call jumpLogin();//不立即跳转(call.interceptor().join(Call).jump())
	}
